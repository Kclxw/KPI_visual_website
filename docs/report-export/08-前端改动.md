### 2.1 src/api/ifir.ts 新增

```typescript
export interface ReportDownloadResult {
  blob: Blob
  filename?: string
}

/**
 * IFIR Model分析报告下载
 */
export async function downloadIfirModelReport(params: {
  start_month: string
  end_month: string
  models: string[]
  segments?: string[]
  odms?: string[]
  tgt?: number
}): Promise<ReportDownloadResult> {
  const request = {
    time_range: { start_month: params.start_month, end_month: params.end_month },
    filters: {
      models: params.models,
      segments: params.segments,
      odms: params.odms
    },
    tgt: params.tgt
  }
  const response = await apiClient.post('/ifir/report/model', request, {
    responseType: 'blob',
    timeout: 120000   // 报告导出允许更长超时
  })
  return {
    blob: response.data,
    filename: getFilenameFromDisposition(response.headers['content-disposition'])
  }
}

/**
 * IFIR ODM分析报告下载
 */
export async function downloadIfirOdmReport(params: {
  start_month: string
  end_month: string
  odms: string[]

  tgt?: number
```

### 3.3 导出逻辑（以 IFIR Model 页为例）

```typescript
import { Download } from '@element-plus/icons-vue'
import { downloadIfirModelReport } from '@/api/ifir'

const exporting = ref(false)

const handleExportReport = async () => {
  if (!dateRange.value || selectedModels.value.length === 0) return
  
  exporting.value = true
  try {
    const { blob, filename } = await downloadIfirModelReport({
      start_month: dateRange.value[0],
      end_month: dateRange.value[1],
      models: selectedModels.value,
      segments: selectedSegments.value.length > 0 ? selectedSegments.value : undefined,
      odms: selectedOdms.value.length > 0 ? selectedOdms.value : undefined,
      tgt: tgtValue.value,
    })

    const fallbackName = buildReportFilename(
      'IFIR',
      'Model',
      selectedModels.value,
      dateRange.value
    )
    await ensureReportBlobOrThrow(blob)
    downloadBlob(blob, filename || fallbackName)
    
    ElMessage.success('报告生成成功')
  } catch (error) {
    console.error('报告生成失败:', error)
    ElMessage.error('报告生成失败，请重试')
  } finally {
    exporting.value = false
  }
}
```

### 3.4 通用下载工具函数（可选提取）

为避免6个页面重复代码，可提取公共工具函数：

```typescript
// src/utils/download.ts

export function downloadBlob(blob: Blob, filename: string) {
  const url = URL.createObjectURL(blob)
  const link = document.createElement('a')
  link.href = url
  link.download = sanitizeFilename(filename)
  document.body.appendChild(link)
  link.click()
  document.body.removeChild(link)
  URL.revokeObjectURL(url)
}

export function sanitizeFilename(filename: string) {
  return filename
    .replace(/[\\\\/:*?\"<>|]/g, '_')
    .replace(/\\s+/g, ' ')
    .slice(0, 180)
}

export function getFilenameFromDisposition(contentDisposition?: string) {
  if (!contentDisposition) return undefined
  const utf8Match = contentDisposition.match(/filename\\*=UTF-8''([^;]+)/i)
  if (utf8Match?.[1]) return decodeURIComponent(utf8Match[1])
  const plainMatch = contentDisposition.match(/filename=\"?([^\";]+)\"?/i)
  return plainMatch?.[1]
}

export function buildReportFilename(
  kpi: 'IFIR' | 'RA',
  dimension: 'Model' | 'ODM' | 'Segment',
  entities: string[],
  dateRange: [string, string]
): string {
  const entitiesStr = entities.length <= 3 
    ? entities.join('+') 
    : `${entities.slice(0, 2).join('+')}等${entities.length}个`
  const range = `${dateRange[0]}至${dateRange[1]}`
  const date = new Date().toISOString().slice(0, 10).replace(/-/g, '')
  return `${kpi}_${dimension}分析报告_${entitiesStr}_${range}_${date}.xlsx`
}
```

## 六、错误处理

后端可能返回 JSON / Problem JSON / HTML 错误页，而不是 Excel 文件，需要在前端统一处理：

```typescript
export async function ensureReportBlobOrThrow(blob: Blob) {
  const type = (blob.type || '').toLowerCase()
  const isExcel = type.includes(
    'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
  )
  const isJsonLike = type.includes('json') || type.includes('text/html')

  if (isExcel) return
  if (!isJsonLike && blob.size > 0) return

  const text = await blob.text()
  try {
    const error = JSON.parse(text)
    throw new Error(error.message || '报告生成失败')
  } catch {
    throw new Error(text || '报告生成失败')
  }
}
```

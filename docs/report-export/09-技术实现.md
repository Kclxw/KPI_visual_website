## 二、report_chart.py - 图表生成模块

### 2.1 模块职责

纯粹负责将数据 → matplotlib 图表 → PNG bytes。不依赖数据库、不依赖FastAPI。

### 2.2 核心函数签名

### 2.3 趋势图实现要点

```python
def generate_trend_chart(entities, tgt=None, value_label="IFIR", 
                         title=None, figsize=(10, 5), dpi=150):
    fig, ax = plt.subplots(figsize=figsize, dpi=dpi)
    
    # 收集所有月份
    all_months = sorted(set(
        p["month"] for e in entities for p in e["trend"]
    ))
    
    # 绘制每个实体的趋势线
    for i, entity in enumerate(entities):
        color = COLORS[i % len(COLORS)]
        month_map = {p["month"]: p["value"] * 1_000_000 for p in entity["trend"]}
        values = [month_map.get(m) for m in all_months]
        ax.plot(all_months, values, '-o', color=color, label=entity["name"],
                linewidth=2, markersize=4)
    
    # TGT 目标线
    if tgt is not None:
        ax.axhline(y=tgt, color=TGT_COLOR, linestyle='--', linewidth=1.5, label=f'TGT ({tgt:,})')
    
    # 样式设置
    ax.set_ylabel(f'{value_label} (DPPM)')
    ax.legend(loc='upper right', fontsize=8)
    ax.grid(True, alpha=0.3)
    ax.tick_params(axis='x', rotation=45 if len(all_months) > 8 else 0)
    if title:
        ax.set_title(title, fontsize=12, fontweight='bold')
    
    fig.tight_layout()
    
    # 输出为 bytes
    buf = BytesIO()
    fig.savefig(buf, format='png', bbox_inches='tight')
    plt.close(fig)
    buf.seek(0)
    return buf.getvalue()
```

### 2.4 饼图实现要点

```python
def generate_pie_chart(items, value_label="IFIR", title=None,
                       figsize=(8, 6), dpi=150):
    fig, ax = plt.subplots(figsize=figsize, dpi=dpi)
    
    names = [item["name"] for item in items]
    values = [item["value"] for item in items]
    colors = [COLORS[i % len(COLORS)] for i in range(len(items))]
    
    wedges, texts, autotexts = ax.pie(
        values, labels=names, colors=colors,
        autopct='%1.1f%%', pctdistance=0.85,
        wedgeprops=dict(width=0.3, edgecolor='white')  # 环形
    )
    
    if title:
        ax.set_title(title, fontsize=12, fontweight='bold')
    
    fig.tight_layout()
    
    buf = BytesIO()
    fig.savefig(buf, format='png', bbox_inches='tight')
    plt.close(fig)
    buf.seek(0)
    return buf.getvalue()
```

### 2.5 中文字体支持

matplotlib 默认不支持中文，需要配置。这里要区分两件事：
- **UTF-8** 负责保证 Python 源文件和字符串编码正确
- **中文字体文件** 负责保证图表能真正渲染出中文字形

推荐使用“项目内置字体文件”而不是依赖操作系统字体路径：

```python
from pathlib import Path

FONT_PATH = (
    Path(__file__).resolve().parents[1]
    / "assets"
    / "fonts"
    / "NotoSansCJKsc-Regular.otf"
)

def _setup_chinese_font():
    """配置中文字体支持"""
    if FONT_PATH.exists():
        fm.fontManager.addfont(str(FONT_PATH))
        plt.rcParams["font.family"] = fm.FontProperties(
            fname=str(FONT_PATH)
        ).get_name()
    else:
        # 允许回退，但必须记录日志告警，避免部署环境静默乱码
        plt.rcParams["font.family"] = "DejaVu Sans"
    
    plt.rcParams["axes.unicode_minus"] = False  # 负号显示

_setup_chinese_font()
```

部署要求：
- 所有新增代码文件统一保存为 **UTF-8 编码**
- 字体文件随服务一起发布，不依赖宿主机手工安装
- CI / 部署启动时检查字体文件是否存在

## 三、report_excel.py - Excel构建模块

### 3.1 模块职责

将分析数据 + 图表PNG → openpyxl Workbook → 临时 `.xlsx` 文件

### 3.2 核心类

```python
"""
报告Excel构建模块
使用 openpyxl 生成多Sheet Excel 并嵌入图表图片
"""
from io import BytesIO
from datetime import datetime
from typing import List, Dict, Optional, Tuple
from openpyxl import Workbook
from openpyxl.styles import Font, Alignment, PatternFill, Border, Side
from openpyxl.utils import get_column_letter
from openpyxl.drawing.image import Image as XlImage


# 样式常量
HEADER_FONT = Font(bold=True, size=11, color="FFFFFF")
HEADER_FILL = PatternFill(start_color="4472C4", end_color="4472C4", fill_type="solid")
TITLE_FONT = Font(bold=True, size=16)
SUBTITLE_FONT = Font(bold=True, size=12)
THIN_BORDER = Border(
    left=Side(style='thin'),
    right=Side(style='thin'),
    top=Side(style='thin'),
    bottom=Side(style='thin')
)


class ReportExcelBuilder:
    """报告Excel构建器"""
    
    def __init__(self, kpi_type: str, dimension: str):
        """
        参数:
            kpi_type: "IFIR" 或 "RA"
            dimension: "Model" 或 "ODM" 或 "Segment"
        """
        self.kpi_type = kpi_type
        self.dimension = dimension
        self.wb = Workbook()
        self.wb.remove(self.wb.active)  # 移除默认sheet
    
    def add_info_sheet(self, meta: dict, filters: dict, tgt: int):
        """Sheet 1: 报告信息"""
    
    def add_trend_sheet(self, entities: list, chart_png: bytes):
        """Sheet 2: 趋势数据 + 嵌入图表"""
    
    def add_top_issue_sheet(self, cards_data: list):
        """Sheet: Top Issue汇总 (Model维度)"""
    
    def add_monthly_top_issue_sheet(self, cards_data: list):
        """Sheet: 月度Top Issue (Model维度)"""
    
    def add_top_model_sheet(self, cards_data: list):
        """Sheet: Top Model汇总 (ODM/Segment维度)"""
    
    def add_monthly_top_model_sheet(self, cards_data: list):
        """Sheet: 月度Top Model"""
    
    def add_top_odm_sheet(self, cards_data: list):
        """Sheet: Top ODM汇总 (Segment维度)"""
    
    def add_monthly_top_odm_sheet(self, cards_data: list):
        """Sheet: 月度Top ODM"""
    
    def add_detail_sheet(self, detail_rows: list, total: int, truncated: bool):
        """Sheet: Detail明细数据"""
    
    def add_comparison_sheet(self, pie_data: list, chart_png: Optional[bytes] = None):
        """Sheet: 多实体对比 + 饼图"""
    
    def save_to_file(self, file_path: str) -> str:
        """构建并保存到文件，返回文件路径"""
        self.wb.save(file_path)
        return file_path
```

## 四、Service层报告生成流程（以IFIR Model为例）

```python
def generate_model_report(self, request: IfirModelReportRequest) -> tuple[str, str]:
    from tempfile import NamedTemporaryFile
    from app.services.report_chart import generate_trend_chart, generate_pie_chart
    from app.services.report_excel import ReportExcelBuilder
    
    # 1. 获取分析数据（复用已有方法）
    analyze_request = IfirModelAnalyzeRequest(
        time_range=request.time_range,
        filters=request.filters,
    )
    analyze_data = self.analyze_model(analyze_request)
    
    # 2. 获取Detail导出数据
    start_date = self._parse_month(request.time_range.start_month)
    end_date = self._parse_month(request.time_range.end_month)
    
    detail_rows, detail_total, truncated = self._get_detail_for_report(
        start_date, end_date,
        models=request.filters.models,
        segments=request.filters.segments,
        odms=request.filters.odms,
    )
    
    # 3. 生成图表
    trend_entities = [
        {"name": card.model, "trend": [{"month": t.month, "value": t.ifir} for t in card.trend]}
        for card in analyze_data.cards
    ]
    trend_png = generate_trend_chart(
        entities=trend_entities,
        tgt=request.tgt,
        value_label="IFIR",
        title="IFIR Model 趋势对比"
    )
    
    pie_png = None
    if analyze_data.summary and analyze_data.summary.model_pie:
        pie_items = [
            {"name": p.model, "value": p.ifir * 1_000_000, "share": p.share}
            for p in analyze_data.summary.model_pie
        ]
        pie_png = generate_pie_chart(
            items=pie_items,
            value_label="IFIR",
            title="Model IFIR 占比"
        )
    
    # 4. 构建Excel
    builder = ReportExcelBuilder(kpi_type="IFIR", dimension="Model")
    
    builder.add_info_sheet(
        meta={
            "data_as_of": analyze_data.meta.data_as_of,
            "time_range": f"{request.time_range.start_month} ~ {request.time_range.end_month}",
            "entities": ", ".join(request.filters.models),
            "segments": ", ".join(request.filters.segments) if request.filters.segments else "全部",
            "odms": ", ".join(request.filters.odms) if request.filters.odms else "全部",
        },
        filters=request.filters.dict(),
        tgt=request.tgt,
    )
    
    builder.add_trend_sheet(
        entities=trend_entities,
        chart_png=trend_png,
    )
    
    builder.add_top_issue_sheet(
        cards_data=[card.dict() for card in analyze_data.cards]
    )
    
    builder.add_monthly_top_issue_sheet(
        cards_data=[card.dict() for card in analyze_data.cards]
    )
    
    builder.add_detail_sheet(detail_rows, detail_total, truncated)
    
    if pie_png:
        pie_table_data = [
            {"model": p.model, "ifir_dppm": round(p.ifir * 1_000_000),
             "share": p.share, "box_claim": p.box_claim, "box_mm": p.box_mm}
            for p in analyze_data.summary.model_pie
        ]
        builder.add_comparison_sheet(pie_table_data, pie_png)
    
    filename = build_report_filename(
        kpi_type="IFIR",
        dimension="Model",
        entities=request.filters.models,
        start_month=request.time_range.start_month,
        end_month=request.time_range.end_month,
    )

    with NamedTemporaryFile(prefix="ifir-model-", suffix=".xlsx", delete=False) as tmp:
        file_path = builder.save_to_file(tmp.name)

    return file_path, filename
```

实现约束：
- `_get_detail_for_report` 必须复用现有 `_apply_detail_segment_filter`
- 涉及 `odms` 时必须复用现有 `_resolve_detail_plants`
- 不允许新写一套只依赖 `map_odm_to_plant` 的静态过滤逻辑
- Detail 数据应尽量边查询边写入 Sheet，避免在内存中长期保留超大 `list[dict]`
